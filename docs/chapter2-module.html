// ------ 関数スコープを使って、外部からアクセスできないオブジェクトを作る
var counter = (function() {
  // プライベート変数
  var count = 0;

  return {
    getCount: function() {
      return count;
    },
    addCount: function(num) {
      count = count + num;
      return count;
    },
    incrementCount: function() {
      count = count + 1;
      return count;
    },
    resetCount: function() {
      count = 0;
      return count;
    },
  };
})();


console.log(counter.getCount());

console.log(counter.incrementCount());

console.log(counter.addCount(3));

console.log(counter.resetCount());

console.log(counter.getCount());
